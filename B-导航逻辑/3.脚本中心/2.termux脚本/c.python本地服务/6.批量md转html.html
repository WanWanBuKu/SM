def md_to_html(md_text: str) -> str:
    """md â†’ å®Œæ•´html å­—ç¬¦ä¸²"""
    md = markdown.Markdown(extensions=["extra", "codehilite", "toc"])
    content = md.convert(md_text)
    title = md.Meta.get("title", [""])[0] if hasattr(md, "Meta") else ""
    if not title:
        # å–ç¬¬ä¸€è¡Œ#æ ‡é¢˜æˆ–æ–‡ä»¶å
        title = content.splitlines()[0].replace("<h1>", "").replace("</h1>", "").strip()

<html lang="zh-CN">
<head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width,initial-scale=1">

    {CSS}
</head>

    {content}
    <hr>
    <small>Generated by å‡‰å®‰ @ {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}</small>
